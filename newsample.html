Basic Data Analysis with Pandas

import pandas as pd import matplotlib.pyplot as plt # Load and explore data df = pd.read_csv('sales_data.csv') print(df.head()) # Basic statistics print(df.describe()) # Visualization df.groupby('month')['sales'].sum().plot(kind='bar') plt.title('Monthly Sales') plt.xlabel('Month') plt.ylabel('Sales ($)') plt.show()
